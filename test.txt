CREATE TENSOR x(1000,14) FROM RANDOM((1000,14),(0,1))
CREATE TENSOR w(1,14) FROM RANDOM((1,14),(0,1),'int') WITH GRAD
CREATE TENSOR y(1000,1)
CREATE TENSOR learning_rate(1,) FROM 0.01
LOOP 5{
SELECT 1/(1+POW(CONSTANT.E, w * x)) AS hx FROM w, x
SELECT y*LOG(hx)+(1-y)(1-hx) AS loss FROM y, hx
SELECT GRADIENT(loss, w) AS g FROM loss, w
SELECT learning_rate*g+w AS w FROM learning_rate, g, w
IF(loss<0.01){
BREAK
}
}
